weight_distribution<-function(g,cumulative = FALSE,...) {
  if (!is_igraph(g)) {
    stop("Not a graph object")
  }
  cs <- strength(g,...)
  hi <- hist(cs, -1:max(cs), plot = FALSE)$density
  if (!cumulative) {
    res <- hi
  }
  else {
    res <- rev(cumsum(rev(hi)))
  }
  res
}
