graph_attr_from_df<-function(g,df) {
  df=as.data.frame(df)
  for (i in 1:ncol(df)) {
    if (class(df[,i])=="factor") df[,i]=as.character(df[,i])
    g=set_graph_attr(g,names(df)[i],df[1,i])
  }
  g
}
